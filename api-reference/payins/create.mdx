---
title: 'Criar Pagamento'
api: 'POST /payin'
description: 'Gera um novo pedido com QR Code Pix ou Boleto.'
---

## Body Parameters

<ParamField body="paymentMethod" type="string" required>
  O método de pagamento escolhido.
  Valores aceitos: `PIX`, `CREDIT_CARD`, `BOLETO`.
</ParamField>

<ParamField body="amount" type="integer" required>
  O valor do pedido em centavos (ex: 10050 para R$ 100,50).
</ParamField>

<ParamField body="referenceId" type="string" required>
  Um identificador único do seu sistema para este pedido.
</ParamField>

<ParamField body="webhookUrl" type="string">
  URL que receberá notificações de mudança de status. Veja o guia de [Webhooks](/webhooks).
</ParamField>

<ParamField body="customer" type="object" required>
  Dados do cliente pagador.
  <Expandable title="Propriedades do Customer">
    <ParamField body="name" type="string" required>Nome completo do cliente.</ParamField>
    <ParamField body="email" type="string" required>Email do cliente.</ParamField>
    <ParamField body="document" type="string" required>CPF ou CNPJ do cliente.</ParamField>
    <ParamField body="phone" type="string">Telefone do cliente.</ParamField>
    <ParamField body="address" type="object">
      Endereço do cliente.
      <Expandable title="Address">
         <ParamField body="street" type="string">Rua.</ParamField>
         <ParamField body="number" type="string">Número.</ParamField>
         <ParamField body="neighborhood" type="string">Bairro.</ParamField>
         <ParamField body="city" type="string">Cidade.</ParamField>
         <ParamField body="state" type="string">Estado.</ParamField>
         <ParamField body="zipCode" type="string">CEP.</ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="items" type="array">
  Lista de itens do pedido.
  <Expandable title="Item Object">
    <ParamField body="title" type="string">Nome do item.</ParamField>
    <ParamField body="quantity" type="number">Quantidade.</ParamField>
    <ParamField body="unitPrice" type="number">Preço unitário em centavos.</ParamField>
    <ParamField body="tangible" type="boolean">Se é um produto físico.</ParamField>
  </Expandable>
</ParamField>

<ResponseField name="id" type="string">
  O ID único do pedido gerado pela plataforma.
</ResponseField>

<ResponseField name="externalId" type="string">
  ID da transação no gateway processador.
</ResponseField>

<ResponseField name="referenceId" type="string">
  Seu ID de referência.
</ResponseField>

<ResponseField name="status" type="string">
  Status inicial do pedido.
</ResponseField>

<ResponseField name="pix" type="object">
  Dados do Pix (qrcode).
</ResponseField>
